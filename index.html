let array = [1, 2, 3, 4];
let sum = 0;

sum += array[i];

2. The second answer is calculated using the reduce array method.

// reduce addUP
const numbers = [1, 2, 3, 4];
function sum1( c, n) {
return c + n;
}
let sum2 = numbers.reduce(sum1);
console.log(sum2);
